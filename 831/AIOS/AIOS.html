from xgboost import XGBRegressor
from sklearn.metrics import mean_squared_error, r2_score, mean_absolute_percentage_error

# Initialize XGBoost regressor without early stopping
reg_xg = XGBRegressor(
    n_estimators=500,
    learning_rate=0.05,
    max_depth=5,
    subsample=0.8,
    colsample_bytree=0.8,
    alpha=0.1,
    reg_lambda=1.5,
    eval_metric="rmse"
)

# Train in a loop to implement early stopping manually
best_mse = float("inf")
best_r2 = float("-inf")
best_mape = float("inf")
best_iteration = 0

for i in range(1, 501):  # Limit to n_estimators
    reg_xg.n_estimators = i
    reg_xg.fit(X_train, y_train)

    # Get predictions and calculate validation metrics
    y_pred_val = reg_xg.predict(X_test)
    mse = mean_squared_error(y_test, y_pred_val)
    r2 = r2_score(y_test, y_pred_val)
    mape = mean_absolute_percentage_error(y_test, y_pred_val)

    # Check if this is the best MSE and update best iteration
    if mse < best_mse:
        best_mse = mse
        best_r2 = r2
        best_mape = mape
        best_iteration = i
    else:
        # Stop training if no improvement for 50 rounds
        if i - best_iteration >= 50:
            break

print("Best iteration:", best_iteration)
print("Best MSE on validation set:", best_mse)
print("Best R2 on validation set:", best_r2)
print("Best MAPE on validation set:", best_mape)
